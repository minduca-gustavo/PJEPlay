// ── TRECHO 1 — Substituir rota_abrirAssistente() ─────────────
//
// Nova versão: passa pjerota_execucao e pjerota_tarefa na URL.
// O assistente.js usa esses parâmetros para validar e preencher
// o cabeçalho.

function rota_abrirAssistente(idTarefa = '', execucao = '') {
    const geo = rota_geometriaModoAssistido()

    let url = extensao_raiz('rota/assistente/assistente.html')
    url += '?pjerota_execucao=' + encodeURIComponent(execucao)
    url += '&pjerota_tarefa='   + encodeURIComponent(idTarefa)

    const assistente = window.open(
        url,
        'rota-pje-assistente',
        'width='   + geo.assistente.width  +
        ',height=' + geo.assistente.height +
        ',left='   + geo.assistente.left   +
        ',top='    + geo.assistente.top    +
        ',toolbar=0,menubar=0,resizable=1'
    )
    armazenar({ rotaGeometria: geo.assistente })
    if (assistente) _rota_janelasMundo.push(assistente)
    return assistente
}


// ── TRECHO 2 — Adicionar após rota_abrirAssistente() ─────────
//
// abrirUrl(url, posicao, nomeJanela)
//
// Abre uma URL numa posição nomeada, calculando a geometria
// automaticamente. Usado pelos roteiros do assistente para
// abrir páginas do PJe sem precisar conhecer os internos
// do pipeline.
//
// posicao: 'esquerda' | 'direita' | 'esquerdaAssistida' |
//          'direitaAssistente' | 'telaCheia'
//
// nomeJanela: nome da janela para window.open (opcional).
//   Se omitido, usa '_blank'.
//   Usar nome fixo permite reusar a janela (foca se já aberta).
//
// Exemplo no roteiro do assistente:
//   abrirUrl(
//     location.origin + '/pjekz/processo/' + id + '/retificar',
//     'esquerdaAssistida',
//     'rota-janela-acao'
//   )

function abrirUrl(url, posicao = 'esquerda', nomeJanela = '_blank') {
    const geo = rota_calcularGeometria(posicao)
    const w = window.open(
        url,
        nomeJanela,
        'width='   + geo.width  +
        ',height=' + geo.height +
        ',left='   + geo.left   +
        ',top='    + geo.top    +
        ',toolbar=0,menubar=0,resizable=1'
    )
    if (w) _rota_janelasMundo.push(w)
    return w
}


// ── TRECHO 3 — Novo nome de janela por execucao ───────────────
//
// Gera um nome único de janela para uma dada execução e tipo.
// Evita conflito entre execuções diferentes abertas ao mesmo tempo.
//
// Exemplo:
//   rota_nomeJanela('detalhe', '1716300000000')
//   → 'rota-detalhe-1716300000000'

function rota_nomeJanela(tipo, execucao) {
    return 'rota-' + tipo + '-' + execucao
}
